def get_divisors(n):
    div = []
    i = 1
    while i <= math.sqrt(n):
        if (i * i) == n:
            div.append(i)
        elif (n % i) == 0:
            div.extend([i, n / i])
        i += 1
    return div

def solution(n):
    divs = get_divisors(n)
    return sum(divs)
