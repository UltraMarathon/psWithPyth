def solution(n):
    answer = 0
    serial = list(range(1, n + 1))
    tmp = 0
    i = 0
    while len(serial):
        tmp += serial[i]
        i += 1
        if tmp >= n:
            if tmp == n:
                answer += 1   
            serial.pop(0)
            i = 0
            tmp = 0
    return answer
