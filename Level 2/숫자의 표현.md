def solution(n):
    cnt = 0
    sum = 0
    head = 1
    for i in range(1, n):
        sum = 0
        for j in range(head, n):
            sum += j
            if(sum == n):
                cnt += 1
                head += 1
                break
            elif(sum > n):
                head += 1
                break
            elif(head == n):
                return 1
    return cnt+1

def solution(n):
    answer = 0
    serial = list(range(1, n + 1))
    tmp = 0
    i = 0
    while len(serial):
        tmp += serial[i]
        i += 1
        if tmp >= n:
            if tmp == n:
                answer += 1   
            serial.pop(0)
            i = 0
            tmp = 0
    return answer
