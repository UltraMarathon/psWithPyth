def solution(s):
    answer = True
    lvl = 0;
    if(s.count('(') == s.count(')')):
        for i in range(len(s)):
            if(i != len(s)-1):
                if(s[0] == ')'):
                    answer = False
                    return answer
                if(s[i] == '(' and s[i+1] == '('):
                    lvl += 1
                    continue
                elif(lvl != 0 and s[i] == ')' and s[i+1] == ')'):
                    lvl -= 1
                    continue
                elif(lvl == 0 and s[i] == s[i+1]):
                    answer = False
                    return answer
    else:
        answer = False

    return answer

def solution(brackets):
    stack = []
    for bracket in brackets:
        if bracket == '(':
            stack.append('(')
        elif bracket == ')' and stack:
            stack.pop()
        else:
            return False
    return not bool(stack)
