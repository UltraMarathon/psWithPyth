# n 번째 계단에 올라가기 위한 경우의 수
# n-1 번째 계단에서 1칸 올라가는 방법 + n-2 번째 계단에서 2칸 올라가는 방법

def solution(n):
    tmp = [1, 1]
    for i in range(2, n + 1):
        tmp.append(tmp[-1] + tmp[-2])
    return tmp[-1] % 1234567
