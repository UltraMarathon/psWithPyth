from collections import deque

def solution(cacheSize, cities):
    answer = 0
    tmp = deque()
    for i in cities:
        i = i.lower()
        if i in tmp:
            answer += 1
            tmp.remove(i)
            tmp.append(i)
        else:
            answer += 5
            j = len(tmp)
            if j + 1 > cacheSize and tmp:
                tmp.popleft()
            if j  <= cacheSize and cacheSize > 0:
                tmp.append(i)
    return answer
