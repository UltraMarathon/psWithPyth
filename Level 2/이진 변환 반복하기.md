def solution(s):
    answer = []
    convert = 0
    removed = 0
    while True:
        removed += s.count('0')
        newstr = s.count('1')
        s = bin(newstr)[2:]
        convert += 1
        if newstr == 1:
            break
    answer.append(convert)
    answer.append(removed)
    return answer
